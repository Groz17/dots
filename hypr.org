#+begin_src bash :tangle ~/.config/hypr/scripts/theme-picker :shebang "#!/bin/bash"
colorscheme=$(gsettings get org.gnome.desktop.interface color-scheme)
# Strip quotes and 'prefer-' prefix
colorscheme=${colorscheme//\'/}
colorscheme=${colorscheme#prefer-}

# Create temporary Lua script
cat > /tmp/theme-picker.lua << EOF
local current_theme = vim.g.colors_name
local theme_list = EfThemes.list.${colorscheme}
local items = {}

for _, theme in ipairs(theme_list) do
  if theme ~= current_theme then
    items[#items + 1] = { text = theme }
  end
end

-- Open picker with ef-themes matching current mode
Snacks.picker.pick({
  items = items,
  source = 'ef_themes_${colorscheme}',
  format = 'text',
  preview = 'colorscheme',
  layout = {fullscreen = true,},
  on_close = function() vim.schedule(function() vim.cmd'q'end)end,
  confirm = function(picker, item)
    picker:close()
    if item then
      vim.schedule(function()
        vim.cmd('colorscheme ' .. item.text)
        vim.fn.writefile({ vim.g.colors_name }, '/tmp/nvim_theme')
      end)
    end
  end,
})
EOF

# Launch nvim and source the script after plugins are loaded
ghostty -e nvim -c "luafile /tmp/theme-picker.lua"

# Read the selected theme
if [ -f /tmp/nvim_theme ]; then
    theme=$(cat /tmp/nvim_theme)
    sed -i "/^\s\\+$colorscheme)/s/\"[^\"]\\+/\"$theme/" ~/'dots/_lib|.org'
    # TODO: Org files that reference theme block should be tangled
    # maybe don't modify org files but just the tangled ones
    ~/dots/tangle ~/dots/_ghostty.org
    find $XDG_RUNTIME_DIR -maxdepth 1 -name 'nvim.*.0' -type s -exec nvim -u NONE --headless --server {} --remote-send "<cmd>colo $theme<cr>" \;
fi
#+end_src
